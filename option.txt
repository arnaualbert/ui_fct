    function cloneFieldset(fieldset, id) {
    // clone the fieldset
    var clone = fieldset.cloneNode(true);
    
    // get all the labels, inputs, and selects in the cloned fieldset
    var labels = clone.querySelectorAll('label');
    var inputs = clone.querySelectorAll('input, select');
    
    // loop over the labels, inputs, and selects, and assign a new id and name to each one
    labels.forEach(label => {
        label.id += '-' + id;
    });
    inputs.forEach(input => {
        input.id += '-' + id;
        input.name += '-' + id;
    });
    
    // return the cloned fieldset
    return clone;
}
